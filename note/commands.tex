% 2. 算子定义 (Operator Definitions)

% -- 常微分
\newcommand{\dd}[1]{\mkern2mu \mathrm{d} #1}

% -- 导数 (Derivative) --
\newcommand{\ode}[3][1]{%
    \ifnum#1=1
        \frac{\dd{#2}}{\dd{#3}}%
    \else
        \frac{\dd{^{#1} #2}}{\dd{#3^{#1}}}%
    \fi
}

% -- 偏导数 (Partial Derivative) --
\newcommand{\pde}[3][1]{%
    \ifnum#1=1
        \frac{\partial #2}{\partial #3}%
    \else
        \frac{\partial^{#1} #2}{\partial #3^{#1}}%
    \fi
}

% -- 混合偏导数 (Mixed Partial Derivative) --
\newcommand{\mpde}[3]{\frac{\partial^2 #1}{\partial #2 \partial #3}}

% -- 全微分/物质导数 (Material Derivative) --
\newcommand{\matderiv}[1]{\frac{\mathrm{D} #1}{\mathrm{D} \timevar}} % D/Dt = ∂/∂t + u ⋅ ∇

% -- 核心动力学算子 (Core Dynamical Operators) --
\newcommand{\opA}[1]{ -(\vel \cdot \nabla) #1 }           % 平流算子 (Advection Operator)
\newcommand{\opC}[1]{ -2\rotationvec \times #1 }          % 科里奥利算子 (Coriolis Operator)
\newcommand{\opP}{ -\frac{1}{\density} \nabla \pressure } % 压力梯度算子 (Pressure Gradient Operator)

% -- 其他 (Others) --
\newcommand{\opG}[1]{ \mathcal{L}_G #1 }  % 重力外波算子 (Gravity External Wave Operator)
\newcommand{\opR}[1]{ \mathcal{L}_R #1 }  % Rossby波算子 (Rossby Wave Operator)
